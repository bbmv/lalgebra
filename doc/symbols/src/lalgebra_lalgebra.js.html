 55   /**
 56    * Check a two-dimensional array.
 57    *
 58    * @param {Array} array A two-dimensional array
 59    * @return {bollean} Returns true if the array is okay
 60    * @throws Array is undefined.
 61    * @throws Array has zero dimension.
 62    * @throws Array has a different amount of elements in the rows.
 63    */
 64   function check2dArray(array) {
 65     var rows, cols, colsTmp, newArr;
 66 
 67     if(!(array instanceof Array)) {
 68       throw errMsg("Array is undefined!", "la-001");
 69     }
 70     rows = array.length;
 71     if(rows===0) {
 72       throw errMsg("Array has zero dimension!", "la-002");
 73     }
 74     
 75     for(var i=0; i<rows; i+=1) {
 76       cols = array[i].length;
 77       if(!colsTmp) colsTmp = cols;
 78       if(colsTmp!==cols) {
 79         throw errMsg("Array has a different amount of elements in the rows!", "la-003");
 80       }
 81       colsTmp = cols;
 82     }
 83 
 84     for(var i=0; i<rows; i+=1)
 85       for(var j=0; j<cols; j+=1) 
 86         if(!isNumber(array[i][j])) return false;
 87 
 88     return true;
 89   }
 90   /**
 91    * Checks the segment for actual points.
 92    *
 93    * @param {object} segment A segment with two points - p1, p2
 94    * @return {boolean} Returns true if successful
 95    * @throws Segment should have two points.
 96    */
 97   function checkSegment(segment) {
 98 
 99     if(!(segment["p1"] && segment["p2"])) {
100       throw errMsg("Segment should have two points: p1 and p2!", "la-005");
101     }
102     var p1 = segment.p1, p2 = segment.p2;
103 
104     p1.x = parseFloat(p1.x); 
105     p1.y = parseFloat(p1.y); 
106     p1.z = parseFloat(p1.z);
107     p2.x = parseFloat(p2.x); 
108     p2.y = parseFloat(p2.y); 
109     p2.z = parseFloat(p2.z);
110 
111     if(isNaN(p1.x) || isNaN(p1.y) || isNaN(p1.z) || isNaN(p2.x) || isNaN(p2.y) || isNaN(p2.z)) {
112       return false;
113     }
114     return true;
115   }

146   function Matrix(array) {
147     var elems, rows, cols;

149     /**
150      * Initialises elements of the matrix.

155     this.initElems = function(array) {

159     };

161      * Gets elements of the matrix.
162      *
163      * @memberof lalgebra.Matrix
164      * @return {number} The elements of the matrix
165      */
166     this.getElems = function() {
167       return copy2dArray(elems);
168     };

171      *
172      * @memberof lalgebra.Matrix
173      * @return {number} An amount of rows
174      */
175     this.getRows = function() {
176       return rows;
177     };
178     /**
179      * Gets an amount of columns of the matrix.
180      *
181      * @memberof lalgebra.Matrix
182      * @return {number} An amount of columns
183      */
184     this.getCols = function() {
185       return cols;
186     };
187     /**
188      * Gets an element of the matrix.
189      *
190      * @memberof lalgebra.Matrix
191      * @param {number} idxRow An index of the row
192      * @param {number} idxCol An index of the column
193      * @return {number} An element of the matrix
194      */
195     this.getElem = function(idxRow, idxCol) {
196       if(inRange(idxRow, idxCol));
197 
198       return elems[idxRow][idxCol];
199     };
200     /**
201      * Sets a value to the element of the matrix.
202      *
203      * @memberof lalgebra.Matrix
204      * @param {number} val A value for the element 
205      * @param {number} idxRow An index of the row
206      * @param {number} idxCol An index of the column
207      */
208     this.setElem = function(val, idxRow, idxCol) {
209       if(isNumber(val) && inRange(idxRow, idxCol)) {
210         elems[idxRow][idxCol] = val;
211       }
212     };
213     this.initElems(array);
214     /**
215      * Checks an index range.
216      *
217      * @param {number} idxRow An index of the row
218      * @param {number} idxCol An index of the column
219      * @return {boolean} Returns true if the value is in a range
220      * @throws The index out of range.
221      */
222     function inRange(idxRow, idxCol) {
223       if(idxRow>=rows || idxRow<0 || idxCol>=cols || idxCol<0) {
224         throw errMsg("The index out of range!", "la-mx-001");
225       }
226       return true;
227     }

248   /**
249    * Multiplys two matrixes.
250    *
251    * @memberof lalgebra.Matrix
252    * @param {Matrix} matrix Second matrix
253    * @return {Matrix} A result of the multiplication
254    * @throws Attempt to multiply by an undefined matrix.
255    * @throws Attempt to multiply matrixes with inappropriate dimensions.
256    */
257   Matrix.prototype.multiply = function(matrix) { 
258     if(!(matrix instanceof Matrix)) {
259       throw errMsg("Attempt to multiply by an undefined matrix!", "la-mx-002");
260     }
261     var getElem1 = this.getElem;
262     var getElem2 = matrix.getElem;
263 
264     var rowsMx1 = this.getRows();
265     var colsMx1 = this.getCols();
266     var rowsMx2 = matrix.getRows();
267     var colsMx2 = matrix.getCols();
268 
269     if(colsMx1!==rowsMx2) { 
270       throw errMsg("Attempt to multiply a matrix with dimensions "+rowsMx1+"x"+colsMx1+
271                   " by a matrix with dimensions "+rowsMx2+"x"+colsMx2+"!", "la-mx-003");
272     }
273     var arrTmp = get2dArray(rowsMx1);
274 
275     for (var i=0; i<rowsMx1; i+=1)
276       for (var j=0; j<colsMx2; j+=1)
277       {
278         arrTmp[i][j] = 0;
279         for (var k=0; k<colsMx1; k+=1)
280         arrTmp[i][j] += getElem1(i, k) * getElem2(k, j);
281       }
282 
283     return new Matrix(arrTmp);
284   };

323   /**
324    * Gets a submatrix without one row and one column.
325    *
326    * @memberof lalgebra.Matrix
327    * @param {Matrix} idxRow The index of the row
328    * @param {Matrix} idxCol The index of the column
329    * @return {Matrix} A minor of the matrix
330    */
331   Matrix.prototype.getMinor = function(idxRow, idxCol) {
332     var getElem = this.getElem;
333     var rowsMx = this.getRows();
334     var colsMx = this.getCols();
335 
336     var arrTmp = get2dArray(rowsMx-1);
337 
338     for(var i=0, ii=0; i<rowsMx; i+=1) {
339       if(i!==idxRow) {
340         for(var j=0, jj=0; j<colsMx; j+=1) {
341           if(j!==idxCol) {
342             arrTmp[ii][jj] = getElem(i, j);
343             jj+=1;
344           }
345         }
346         ii+=1;
347       }
348     }
349     return new Matrix(arrTmp);
350   };
351   /**
352    * Computes a determinant of the matrix.
353    *
354    * @memberof lalgebra.Matrix
355    * @return {number} A determinant of the matrix
356    * @throws Attempt to transpose a non quadratic matrix.
357    */
358   Matrix.prototype.determinant = function() {
359     var getElem = this.getElem;
360     var rowsMx = this.getRows();
361     var colsMx = this.getCols();
362     var det = 0;
363 
364     if(rowsMx!==colsMx) {
365       throw errMsg("Attempt to transpose a non quadratic matrix!", "la-mx-004");
366     }
367     switch(rowsMx) {
368       case 2: 
369         det = getElem(0, 0) * getElem(1, 1) - getElem(0, 1) * getElem(1, 0);
370         break;
371       default: 
372         for(var j=0; j<colsMx; j++)
373           det = det + getElem(0, j) * Math.pow(-1, 2 + j) * this.getMinor(0, j).determinant();
374     }
375 
376     return det;
377   };    
378   /**
379    * Inverses the matrix.
380    *
381    * @memberof lalgebra.Matrix
382    * @return {Matrix} An inversed matrix
383    * @throws Inverse matrix does not exist.
384    */
385   Matrix.prototype.inverse = function() {
386     var rowsMx = this.getRows();
387     var colsMx = this.getCols();
388     var det = this.determinant();
389     
390     if(det===0) {
391       throw errMsg("Inverse matrix does not exist!", "la-mx-005");
392     }
393 
394     var arrTmp = get2dArray(rowsMx);
395 
396     for(var i=0; i<rowsMx; i+=1)
397       for(var j=0; j<colsMx; j+=1)
398         arrTmp[i][j] = Math.pow(-1, 2 + i + j) * this.getMinor(i, j).determinant() / det;
399 
400     return (new Matrix(arrTmp)).transpose();
401   };  
402   /**
403    * Normolizes the matrix of an object in homogeneous coordinates 
404    * to get a back projection into the actual dimensional space.
405    *
406    * @memberof lalgebra.Matrix
407    * @return {Matrix} An normolized matrix
408    */
409   Matrix.prototype.normolize = function() {
410     var rows = this.getRows();
411     var cols = this.getCols();
412     var getElem = this.getElem;
413     var last;
414 
415     var arrTmp = get2dArray(rows);
416     
417     for (var i=0; i<rows; i++) {
418       for (var j=0; j<cols; j++) {
419         last = getElem(i, cols-1);
420         if(last!=0) arrTmp[i][j] = getElem(i, j) / last, i, j;
421         else arrTmp[i][j] = 0;
422       }
423     }
424     return new Matrix(arrTmp);
425   };
426   /**
427    * Finds an intersection of two segments.
428    *
429    * @memberof lalgebra
430    * @param {object} a1 First point (x,y,z) of first segment (a)
431    * @param {object} a2 Second point (x,y,z) of first segment (a)
432    * @param {object} b1 First point (x,y,z) of second segment (b)
433    * @param {object} b2 Second point (x,y,z) of second segment (b)
434    * @return {object} An intersection of two segments (x,y,z)
435    */
436   function getIntersection(a1, a2, b1, b2) {
437     var inter = {};
438 
439     if(tripleProduct({"x":a2.x-a1.x, "y":a2.y-a1.y, "z":a2.z-a1.z}, 
440       {"x":b2.x-b1.x, "y":b2.y-b1.y, "z":b2.z-b1.z}, 
441       {"x":a2.x-b1.x, "y":a2.y-b1.y, "z":a2.z-b1.z})!=0)  {
442       console.log("Отрезки в разных плоскостях!");
443       return false;
444     }
445 
446     var d1 = (a2.x-a1.x)*(b2.y-b1.y)-(a2.y-a1.y)*(b2.x-b1.x);
447     var d2 = (b2.y-b1.y)*(a2.x-a1.x)-(b2.x-b1.x)*(a2.y-a1.y);
448     if(d1===0 || d2===0) {
449       console.log("Прямые параллельны!");
450       return false;
451     }
452 
453     var t1 = ((b1.x-a1.x)*(b2.y-b1.y) - (b1.y-a1.y)*(b2.x-b1.x)) / d1;
454     var t2 = ((b1.x-a1.x)*(a2.y-a1.y) - (b1.y-a1.y)*(a2.x-a1.x)) / d2;
455     
456     if(t1<0 || t1>1 || t2<0 || t2>1) {
457       console.log("The segments are not intersection!");
458       return false;
459     }
460 
461     inter.x = Math.round(b1.x+(b2.x-b1.x)*t2);
462     inter.y = Math.round(b1.y+(b2.y-b1.y)*t2);
463     inter.z = Math.round(b1.z+(b2.z-b1.z)*t2);
464 
465     return inter;
466   }

534   /**
535    * Computes two angles: 
536    * first - between a vector projection on the plane y=0 and the z-axis,
537    * second - between a vector and the y-axis.
538    * It works for all quadrants.
539    * 
540    * @memberof lalgebra
541    * @param {object} vector An end point (x,y,z) of the vector
542    * @return {object} Two angles
543    */
544   function getAngles(vector) {
545     var x = vector.x;
546     var y = vector.y;
547     var z = vector.z;
548     var radianToDegree = lalgebra.radianToDegree;
549     var angles = {};
550     var k=1;
551     
552     if(x<0) k=-1;
553     if(x!==0 || z!==0) {
554       angles.a = radianToDegree(Math.acos(z/Math.sqrt(x*x+z*z)))*k;
555     } else {
556       angles.a = 0;  
557     }
558     angles.b = radianToDegree(Math.acos(y/Math.sqrt(x*x+y*y+z*z)));
559     return angles;
560   }  
561   /**
562    * Converts Cartesian coordinates into spherical coordinates. 
563    * 
564    * @memberof lalgebra
565    * @param {object} cart An end point (x,y,z) of the vector 
566    * @return {object} Spherical coordinates (r,a,b)
567    */
568   function cartesianToSpherical(cart) {
569     var x, y, z, r, a, b;
570     x = cart.x;
571     y = cart.y;
572     z = cart.z;
573 
574     r = Math.sqrt(x*x + y*y + z*z);
575     if(r===0) { 
576       a = b = 0;
577     } else {
578       b = Math.acos(z/r);
579       if(x===0) {
580         if(y<0) a = Math.PI*3/2;
581         else a = Math.PI/2;
582       } else {
583         a = Math.atan(y/x);
584         if(x<0) a += Math.PI;
585       }
586     }
587     var sph = { "r":r, "a":radianToDegree(a), "b":radianToDegree(b) };
588     return sph;
589   }
