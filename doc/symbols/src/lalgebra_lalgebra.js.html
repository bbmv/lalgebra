 29   /**
 30    * Creates an empty two-dimensional array.
 31    * 
 32    * @param {number} rows Amount of rows
 33    * @return {Array} The empty two-dimensional array
 34    */
 35   function get2dArray(rows) {
 36     var arr = [];
 37     for(var i=0; i<rows; i+=1)
 38       arr[i] = [];
 39     return arr;

 41   /**
 42    * Creates a matrix.
 43    * 
 44    * @memberof lalgebra
 45    * @constructor
 46    * @this {Matrix}
 47    * @param {Array} array An array of elements for the matrix.
 48    * @property {Array} array An array of the matrix elements.
 49    * @property {number} rows An amount of the matrix rows.
 50    * @property {number} cols An amount of the matrix columns.
 51    * @throws Array is undefined.
 52    * @throws Array has zero dimension.
 53    * @throws Array has a different amount of elements in the rows.
 54    */
 55   function Matrix(array) {
 56     var rows, cols, colsTmp;
 57 
 58     if(!array) {
 59       throw errMsg("Array is undefined!", "va-mx-001");
 60     }
 61     rows = array.length;
 62     if(rows===0) {
 63       throw errMsg("Array has zero dimension!", "va-mx-002");
 64     }
 65     for(var i=0; i<rows; i+=1) {
 66       cols = array[i].length;
 67       if(!colsTmp) colsTmp = cols;
 68       if(colsTmp!==cols) {
 69         throw errMsg("Array has a different amount of elements in the rows!", "va-mx-003");
 70       }
 71       colsTmp = cols;
 72     }
 73     this.array = array;
 74     this.rows = rows;
 75     this.cols = cols;
 76   }

 92   /**
 93    * Multiplys two matrixes.
 94    *
 95    * @memberof lalgebra.Matrix
 96    * @param {Matrix} matrix Second matrix
 97    * @return {Matrix} A result of the multiplication
 98    * @throws Attempt to multiply by an undefined matrix.
 99    * @throws Attempt to multiply matrixes with inappropriate dimensions.
100    */
101   Matrix.prototype.multiply = function(matrix) { 
102     if(!(matrix instanceof Matrix)) {
103       throw errMsg("Attempt to multiply by an undefined matrix!", "va-mx-004");
104     }
105     if(this.cols!==matrix.rows) { 
106       throw errMsg("Attempt multiplied a matrix with dimensions "+this.rows+"x"+this.cols+
107                   " by a matrix with dimensions "+matrix.rows+"x"+matrix.cols+"!", "va-mx-005");
108     }
109     var arrTmp = get2dArray(this.rows);
110 
111     for (var i=0; i<this.rows; i+=1)
112       for (var j=0; j<matrix.cols; j+=1)
113       {
114         arrTmp[i][j] = 0;
115         for (var k=0; k<this.cols; k+=1)
116         arrTmp[i][j] += this.array[i][k] * matrix.array[k][j];
117       }
118 
119     return new Matrix(arrTmp);
120   }
121   /**
122    * Transposes the matrix.
123    *
124    * @memberof lalgebra.Matrix
125    * @return {Matrix} A transposed matrix
126    */
127   Matrix.prototype.transpose = function() {
128     var arrTmp = get2dArray(this.cols);
129 
130     for(var i=0; i<this.rows; i+=1)
131       for(var j=0; j<this.cols; j+=1)
132         arrTmp[j][i] = this.array[i][j];
133 
134     return new Matrix(arrTmp);
135   }
136   /**
137    * Copies the matrix.
138    *
139    * @memberof lalgebra.Matrix
140    * @return {Matrix} A copy of the matrix
141    */
142   Matrix.prototype.getCopy = function() {
143     var arrTmp = get2dArray(this.rows);
144 
145     for(var i=0; i<this.rows; i+=1)
146       for(var j=0; j<this.cols; j+=1)
147         arrTmp[i][j] = this.array[i][j];
148 
149     return new Matrix(arrTmp);
150   }
151   /**
152    * Gets a submatrix without one row and one column.
153    *
154    * @memberof lalgebra.Matrix
155    * @param {Matrix} idxRow The index of the row
156    * @param {Matrix} idxCol The index of the column
157    * @return {Matrix} An minor
158    */
159   Matrix.prototype.getMinor = function(idxRow, idxCol) {
160     var tmpMatrix = this.getCopy();
161     tmpMatrix.array.splice(idxRow, 1);
162     tmpMatrix.rows -= 1;
163     tmpMatrix.cols -= 1;
164 
165     for(var i=0; i<tmpMatrix.rows; i++) {
166       tmpMatrix.array[i].splice(idxCol, 1);
167     }
168     return tmpMatrix;
169   }
170   /**
171    * Computes a determinant of the matrix.
172    *
173    * @memberof lalgebra.Matrix
174    * @return {number} A determinant of the matrix
175    * @throws Attempt to transpose a non quadratic matrix.
176    */
177   Matrix.prototype.determinant = function() {
178     var det = 0;
179 
180     if(this.rows!==this.cols) {
181       throw errMsg("Attempt to transpose a non quadratic matrix!", "va-mx-007");
182     }
183     switch(this.rows) {
184       case 2: 
185         det = this.array[0][0] * this.array[1][1] - this.array[0][1] * this.array[1][0];
186         break;
187       default: 
188         for(var j=0; j<this.cols; j++)
189           det = det + this.array[0][j] * Math.pow(-1, 2 + j) * this.getMinor(0, j).determinant();
190     }
191 
192     return det;
193   }     
194   /**
195    * Inverses the matrix.
196    *
197    * @memberof lalgebra.Matrix
198    * @return {Matrix} An inversed matrix
199    * @throws Attempt to inverse a non quadratic matrix.
200    * @throws Inverse matrix does not exist.
201    */
202   Matrix.prototype.inverse = function() {
203     if(this.rows!==this.cols) {
204       throw errMsg("Attempt to inverse a non quadratic matrix!", "va-mx-008");
205     }
206         
207     var det = this.determinant();
208     
209     if(det===0) {
210       throw errMsg("Inverse matrix does not exist!", "va-mx-009");
211     }
212 
213     var arrTmp = get2dArray(this.rows);
214 
215     for(var i=0; i<this.rows; i+=1)
216       for(var j=0; j<this.cols; j+=1)
217         arrTmp[i][j] = Math.pow(-1, 2 + i + j) * this.getMinor(i, j).determinant() / det;
218 
219     return (new Matrix(arrTmp)).transpose();
220   }

263   /**
264    * Computes a magnitude (length) of the vector.
265    *
266    * @memberof lalgebra
267    * @param {object} a The vector
268    * @return {number} A magnitude of the vector
269    */
270   function vectorMagnitude(a) {
271     return  Math.sqrt(a.x*a.x + a.y*a.y + a.z*a.z);
272   }
273   /**
274    * Computes a vector (cross) product of two vectors.
275    *
276    * @memberof lalgebra
277    * @param {object} a First vector
278    * @param {object} b Second vector
279    * @return {object} A vector product of two vectors
280    */
281   function vectorProduct(a, b) {
282     return  { x: a.y*b.z - a.z*b.y,
283               y: a.z*b.x - a.x*b.z,
284               z: a.x*b.y - a.y*b.x };
285   }
286   /**
287    * Computes a scalar (dot) product of two vectors.
288    *
289    * @memberof lalgebra
290    * @param {object} a First vector
291    * @param {object} b Second vector
292    * @return {number} A scalar product of two vectors
293    */
294   function scalarProduct(a, b) {
295     return  a.x*b.x + a.y*b.y + a.z*b.z;
296   }
297   /**
298    * Computes a scalar triple product of three vectors.

300    * @memberof lalgebra

303    * @param {object} c Third vector
304    * @return {number} A scalar triple product of two vectors

306   function tripleProduct(a, b, c) {

309   /**

326     }
